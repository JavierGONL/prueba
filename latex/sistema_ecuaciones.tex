\section*{Sistema de Ecuaciones}

\subsection*{Barra R}

\textbf{Dirección $n$ (normal):}
\begin{equation}
n - R: \left(W_R + r_y + K_y - (\Theta_1)_y \right)\cos(\Gamma) - \left(K_x + r_x - (\Theta_1)_x\right)\sin(\Gamma) = m_R \cdot (\bar{a}_R)_n
\end{equation}

\textbf{Dirección $t$ (tangencial):}
\begin{equation}
t - R: \left((\Theta_1)_y - W_R - r_y - K_y \right)\sin(\Gamma) - \left(K_x + r_x - (\Theta_1)_x\right)\cos(\Gamma) = m_R \cdot (\bar{a}_R)_t
\end{equation}

\textbf{Momento respecto a $G_R$:}
\begin{multline}
\ddot{\theta} - R: -\left| \frac{R}{2} - L \right| \left[\left( r_y\cdot \sin(\Gamma) + \cos(\Gamma) \cdot r_x\right) - \frac{R}{2}\left( K_y \cdot \sin(\Gamma) + K_x \cdot \cos(\Gamma)\right)\right] - \\
- \frac{R}{2}\left( (\Theta_1)_y \cdot \sin(\Gamma) + (\Theta_1)_x \cdot \cos(\Gamma)\right) = \frac{1}{12}m_R \cdot R^2 \cdot \alpha_R
\end{multline}

\subsection*{Barra r (Manivela)}

\textbf{Dirección $n$ (normal):}
\begin{equation}
n - r: \left(W_r + (\Theta_2)_y - r_y\right)\cos(\delta) + \left(r_x - (\Theta_2)_x\right)\sin(\delta) = m_r \cdot (\bar{a}_r)_n
\end{equation}

\textbf{Dirección $t$ (tangencial):}
\begin{equation}
t - r: \left(r_y - W_r - (\Theta_2)_y\right)\sin(\delta) + \left(r_x - (\Theta_2)_x\right)\cos(\delta) = m_r \cdot (\bar{a}_r)_t
\end{equation}

\textbf{Momento respecto a $G_r$:}
\begin{multline}
\ddot{\theta} - r: \frac{r}{2}\left[r_x \cos(\delta) + r_y \sin(\delta) + (\Theta_2)_x \cos(\delta) + (\Theta_2)_y \sin(\delta)\right] = \frac{1}{12}m_r \cdot r^2 \cdot \ddot{\alpha}_r = 0 \quad \rightarrow \quad \ddot{\alpha}_r = 0
\end{multline}

\subsection*{Barra K (Biela)}

\textbf{Dirección $x$:}
\begin{equation}
x - K: K_x - M_x = m_K \cdot (\bar{a}_K)_x
\end{equation}

\textbf{Dirección $y$:}
\begin{equation}
y - K: K_y - W_K - M_y = m_K \cdot (\bar{a}_K)_y
\end{equation}

\textbf{Momento respecto a $G_K$:}
\begin{equation}
\ddot{\theta} - K: \frac{K}{2}\left(K_x \cos(\beta) + K_y \sin(\beta)\right) + M_x \cos(\beta) + M_y \sin(\beta) = \frac{1}{12}m_K \cdot K^2 \cdot \ddot{\alpha}_K
\end{equation}

\subsection*{Martillo M}

\textbf{Dirección $x$:}
\begin{equation}
x - M: M_x = 0
\end{equation}

\textbf{Dirección $y$:}
\begin{equation}
y - M: W_M - M_y = m_M \cdot (\bar{a}_M)_y
\end{equation}

\textbf{Momento:}
\begin{equation}
\ddot{\theta} - M: \text{(no hay rotación)}
\end{equation}
